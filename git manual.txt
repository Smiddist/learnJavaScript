
cd d: - сменить дерикторию
pwd - где я 
ls - чего там еще есть
git clone - клонировать гит
cd .. - выше в каталог 
git status - чего не покамитил 
git init - инициализация гита?
git add <название файла> - добавить файл в комит
git add -u - добавить все измененные файлы в комит(если файлы уже были добавлены ранее)
git add . - добавить все файлы в комит
git commit -m - комит и комент
git checkout -b <название новой ветки> <название ветки с которой нужно клонировать код> - сменить ветку (создать -b)
git branch - какая ветка 
git branch -d - удолить ветку
git branch -M - переименовать ветку
git remote -v - проверить ссылки подключения к гиту
git push <куда пихать(ИМЯ ссылки)> <название пихуемой ветки> - толкнуть на гитхаб
git pill <название ветки>- актуализация ветки из гитхаб
